class Solution:
    def kidsWithCandies(self, candies, extraCandies):
        max_candies = max(candies)
        result = []
        for candy in candies:
            result.append(candy + extraCandies >= max_candies)
        return result

candies = [2,3,5,1,3]
extraCandies = 3
sol = Solution()
print(sol.kidsWithCandies(candies, extraCandies))
